Atlantis

Abridge problem statement

給定平面上 N 個矩形，求這些矩形覆蓋的面積。

AC code

#include <cstdio>
#include <vector>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

// 矩形面積
const int MAX_NN = (1 << 17);

struct Rect{
	double x1, y1, x2, y2;
};

struct Event{
	double y; 
	int x1, x2, type;
	bool operator < (const Event& e) const{
		if (y == e.y)
			return type < e.type;
		return y < e.y;
	}
};

int N;
vector<Rect> rect;
vector<Event> event;
vector<double> xs;

//Segment Tree
struct SegTree{
	int NN;
	int cnt[MAX_NN];
	double len[MAX_NN];

	void init(int n){
		NN = 1;
		while(NN < n){
			NN <<= 1;
		}
		fill(cnt, cnt + 2 * NN, 0);
		fill(len, len + 2 * NN, double(0.0));
	}

	void maintain(int u, int l, int r){
		if (cnt[u] > 0)
			len[u] = xs[r] - xs[l];
		else{
			if (u >= NN - 1)
				len[u] = 0;
			else
				len[u] = len[2 * u + 1] + len[2 * u + 2];
		}
	}

	void update(int a, int b, int x, int u, int l, int r){
		if (a >= r || b <= l)
			return ;
		if (a <= l && r <= b){
			cnt[u] += x;
			maintain(u, l, r);
			return;
		}

		int m = (l + r) / 2;
		update(a, b, x, u * 2 + 1, l, m);
		update(a, b, x, u * 2 + 2, m, r);
		maintain(u, l, r);

	}

};



double solve(){
	//離散化
	xs.clear();
	for (int i = 0; i < N; i++){
		xs.push_back(rect[i].x1);
		xs.push_back(rect[i].x2);
	}
	sort(xs.begin(), xs.end());
	xs.resize(unique(xs.begin(), xs.end()) - xs.begin());
	
	//Sweep line
	event.clear();
	for (int i = 0; i < N; i++){
		int x1 = lower_bound(xs.begin(), xs.end(), rect[i].x1) - xs.begin();
		int x2 = lower_bound(xs.begin(), xs.end(), rect[i].x2) - xs.begin();
		event.push_back((Event) {rect[i].y1, x1, x2, +1});
		event.push_back((Event) {rect[i].y2, x1, x2, -1});
	}
	sort(event.begin(), event.end());
	
	SegTree seg;
	seg.init(xs.size());
	seg.update(event[0].x1, event[0].x2, event[0].type, 0, 0, seg.NN);
	
	double ans = 0.0;
	for (int i = 1; i < event.size(); i++){
		ans += seg.len[0] * (event[i].y - event[i - 1].y);
		seg.update(event[i].x1, event[i].x2, event[i].type, 0, 0, seg.NN);
	}
	return ans;
}

int main(){
	int T = 0;
	
	while (scanf(" %d", &N)){
		if (N == 0)
			break;

		rect.clear();
		for (int i = 0; i < N; i++){
			double x1, y1, x2, y2;
            scanf(" %lf %lf %lf %lf", &x1, &y1, &x2, &y2);
			rect.push_back((Rect) {x1, y1, x2, y2});
		}

		printf("Test case #%d\n", ++T);
		printf("Total explored area: %.2f\n", solve());
		puts("");
	}

	return 0;
}
